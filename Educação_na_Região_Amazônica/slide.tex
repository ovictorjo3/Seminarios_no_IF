	\section{Introdução}
	
	\begin{frame}{Consideremos a seguinte situação}
		\justifying
		Cássio depositou uma certa quantia em uma caderneta de poupança especial, que rende 1\% ao mês. Por quantos meses ele deverá deixar o dinheiro na conta para que seu valor dobre? (Suponha que não sejam feitos retiradas nem novos depósitos.)
	\end{frame}
	
	\begin{frame}{Função Logarítmica}
		\begin{block}{Definição}
			Dado um número real $a \ (0<a\neq1)$, chamamos de \textbf{função logarítmica} de base $\textbf{a}$ a função $f$ de $\mathbb{R_+^*} \ \text{em} \ \mathbb{R}$ que associa a cada $x$ o número $\log_{a}x$.
		\end{block} Exemplos de funções logarítmicas em $\mathbb{R_+^*}$
		
		\begin{itemize}
			\item [a)] $f(x)=\log_{2}x$
			\item [b)] $g(x)=\log_{\frac{1}{2}}x$
			\item [c)] $h(x)=\log x$
			\item [d)] $p(x)=\ln x$
		\end{itemize}
	\end{frame}
	
	\section{Propriedades}
	
	\begin{frame}{Propriedades (P)}
		\begin{block}{P.1}
			Se $0<a\neq 1$, então as funções $f$ de $\mathbb{R_{+}^*} \ \text{em} \ \mathbb{R}$,
			definida por $f(x)=\log_{a}x$, e $g$ de $\mathbb{R} \ \text{em} \ \mathbb{R_{+}^*}$, definida
			por $g(x)=a^{x}$, são inversas uma da outra. 
		\end{block}
	\end{frame}
	
	\begin{frame}{Propriedades (P)}
		
		\begin{block}{P.2}
			A função logarítmica $f(x)=log_{a}x$ é crescente (decrescente) se, e somente se,
			$a>1\  (0<a< 1)$            
		\end{block}
	\end{frame}
	
	\section{Observações}
	
	\begin{frame}{Observações (O)}
		\begin{block}{O.1}
			Quando a base é maior que 1, a relação de desigualdade existente entre os logaritmos
			de dois números positivos tem o mesmo sentido que a relação entres esses números.
		\end{block} Exemplos:
		
		\begin{itemize}
			\item [a)] $4>2 \Rightarrow \log_{2}4 > \log_{2}2$
			\item [b)] $15 >4 \Rightarrow \log_{3}15 > \log_{3}4$
			\item [c)] $0,42 < 6,3 \Rightarrow \log_{7}0,42 < \log_{7}6,3$
			\item [d)] $4 > 0,3 \Rightarrow \ln 4 > \ln 0,3$
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Observações (O)}
		\begin{block}{O.2}
			Quando a base é positiva e menor que 1, a relação de desigualdade existente entre os logaritmos
			de dois números positivos é de sentido contrário à que existe entre esses números.
		\end{block} Exemplos:
		
		\begin{itemize}
			\item [a)] $8>2 \Rightarrow \log_{\frac{1}{2}}8 < \log_{\frac{1}{2}}2$
			\item [b)] $12>5 \Rightarrow \log_{\frac{1}{3}}12 < \log_{\frac{1}{3}}5$
			\item [c)] $\sqrt{3} < 7 \Rightarrow \log_{0,1}\sqrt{3}>\log_{0,1}7$
			\item [d)] $0,3 < 2,4 \Rightarrow \log_{0,2}0,3 > \log_{0,2}2,4$
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Observações (O)}
		\begin{block}{O.3}
			Se a base é maior que 1, então os números positivos menores que 1 têm logaritmos
			negativos e os números maiores que 1 tem logaritmos postivos.
		\end{block} Exemplos:
		
		\begin{itemize}
			\item [a)] $\log_{2}0,25 < 0$
			\item [b)] $\log 0,02 < 0$
			\item [c)] $\log_{2}32 > 0$
			\item [d)] $\log_{3} \sqrt{5} > 0$
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Observações (O)}
		\begin{block}{O.4}
			Se a base é positiva e menor que 1, então os números positivos menores que 1 têm
			logaritmos positivos e os númeors maiores que 1 têm logaritmos negativos.
		\end{block} Exemplos:
		
		\begin{itemize}
			\item [a)] $\log_{0,5}0,25 > 0$
			\item [b)] $\log_{0,1}0,03 > 0$
			\item [c)] $\log_{0,5}4 < 0$
			\item [d)] $\log_{0,2}\sqrt{3} < 0$
		\end{itemize}
	\end{frame}
	
	\section{Imagem}
	
	\begin{frame}{Imagem}
		\justifying
		Se $0<a\neq 1$ então a função $f$ de $\mathbb{R_{+}^*} \ \text{em} \ \mathbb{R}$ definida
		por $f(x)=\log_{a}x$ admite a função inversa de $g$ de $\mathbb{R} \ \text{em} \ \mathbb{R_{+}^*}$ definida
		por $g(x)=a^{x}$. Logo, $f$ é bijetora e, portanto, a imagem de $f$ é: $$\text{Im} = \mathbb{R}$$
	\end{frame}
