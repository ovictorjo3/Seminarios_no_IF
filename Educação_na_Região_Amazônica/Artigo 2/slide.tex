	\section{Opção Decolonial}
	
		\begin{frame}{Opção Decolonial}
			\begin{block}{Desafio histórico da colonialidade na formação docente}
				\justifying
				"Considerar a formação de professoras e professores que ensinam matemática tendo como opção a decolonialidade nos impõe, ainda, o desafio de compreender historicamente o modo como, nos contextos brasileiros, nossos processos formativos são permeados por traços e efeitos da colonialidade."
			\end{block}
		\end{frame}
		
		\begin{frame}{Opção Decolonial}
			\begin{block}{Desestabilização das narrativas hegemônicas}
				\justifying
				"Nossa intenção é apenas indagar essa formação desestabilizando as narrativas hegemônicas que a historiografia da educação brasileira buscou produzir e consolidar, questionando alguns lugares comuns e olhares de cima que caracterizam os estudos sobre a história da formação de professoras e professores que ensinam matemática."
			\end{block}
		\end{frame}
		
		\begin{frame}{Opção Decolonial}
			\begin{block}{Exercício de nomeação e recuperação da autoria histórica}
				\justifying
				Como destacam Ramalho e Porta (2018), trata-se de um exercício de nomear o que não tem nome, de enfraquecer cada vez mais a produção unívoca da história, recuperando o direito de nomear e de decidir como contar nossas histórias.
			\end{block}
		\end{frame}
	

	\section{Critica a colonialidade}
	
		\begin{frame}{Critica a colonialidade}
			\begin{block}{Definição de colonialidade do ser}
				\justifying
				Walsh (2008) demarca a colonialidade do ser como a dimensão ontológica da colonialidade, que se manifesta por meio da inferiorização, da subalternização do sujeito e de suas capacidades cognitivas.
			\end{block}
		\end{frame}
		
		
		\begin{frame}{Critica a colonialidade}
			\begin{block}{Função social da subalternização}
				\justifying
				"A constituição de grupos subalternizados permite que esses ocupem posições sociais e de trabalho que não são escolhidas por eles, de cujas funções sociais eles podem nem mesmo ter consciência, e que servem a um projeto político e econômico de dominação."
			\end{block}
			\uncover<2->{
				\begin{block}{Associação à desumanização e à "não-existência"}
					\justifying
					A colonialidade do ser é, em geral, associada à desumanização de grupos sociais por conta de sua cor ou de suas raízes, descrita por Fanon (2008) como o 'tratamento da nãoexistência'.
				\end{block}
			}
		\end{frame}
		
		\begin{frame}{Critica a colonialidade}
			\begin{block}{Ampliação para outros grupos subalternizados}
				\justifying
				Entretanto, processos de colonialidade do ser também podem ser reconhecidos em outros grupos que são relegados a lugares sociais subalternizados a serviço de um projeto de poder político e econômico, quando as identidades dos sujeitos, enquanto membros desses grupos, são desqualificadas ou inferiorizadas.
			\end{block}
		\end{frame}
	
	\section{Suberverção da narrativa histórica}
	
		\begin{frame}{Suberverção da narrativa histórica}
		
			\justifying
			\textbf{Recusa da versão passiva da história} \\
			Tal mudança na pessoa que narra não corresponde meramente a outra versão da mesma história, em que os povos colonizados contemplam passivamente a aproximação das caravelas carregando os dispositivos civilizatórios que apagarão suas culturas.
				
		\end{frame}
		
		\begin{frame}{Suberverção da narrativa histórica}
			
			\justifying
			\textbf{Contraposição à história heroica e inventada} \\
			Esta não é uma versão da história dos heróis e dos fatos heroicos que atuaram na invenção daquilo que hoje se chama Brasil, daquilo que hoje se chama Matemática ou daquilo que hoje se chama Formação de Professores \\
			\vspace{2mm}
			\uncover<2->{
			\textbf{Protagonismo dos invisibilizados} \\
			Constroi-se uma narrativa de grupos, de sujeitos e de sabedorias que foram e que são invisibilizados nesses processos.
			}

		\end{frame}
		
	\section{Da crítica à insurgência}
	
		\begin{frame}{Da crítica à insurgência}
			\begin{alertblock}{Crítica ao marco fundador da formação (USP, 1934)}
				\justifying
				\textbf{A narrativa hegemônica privilegia centros urbanos e formação universitária.}. "Essa narrativa produz efeitos que mantêm traços da colonialidade nos processos formativos."
			\end{alertblock}
			
			\uncover<2->{
			\begin{alertblock}{Subordinação epistemológica do professor}
				\justifying
				\textbf{Ser professor condicionado a ser matemático acadêmico.} "Não pode haver professor de matemática que desconheça a matemática praticada nos espaços científico-acadêmicos."
			\end{alertblock}
			}
		\end{frame}
		
		\begin{frame}{Da crítica à insurgência}
			\begin{alertblock}{Universidade como núcleo vigilante}
				\justifying
				\textbf{Ela não apenas forma, mas regula e vigia.} "A Universidade se converteu no núcleo vigilante do processo educacional." 
			\end{alertblock}
			
			\uncover<2->{
				\begin{alertblock}{Colonialidade do ser e do saber articuladas}
					\justifying
					\textbf{Professor relegado a posição inferior, desautorizado sobre seu próprio fazer.} "A autoridade sobre os saberes é outorgada a atores externos a seus próprios espaços profissionais." 
				\end{alertblock}
			}
		\end{frame}
		
		\begin{frame}{Da crítica à insurgência}
			\begin{alertblock}{Coletivos docentes como resistência.}
				\justifying
				\textbf{Tomar a primeira pessoa das narrativas.} "Somente a partir do entendimento de suas ações em contextos mais amplos, professoras e professores podem (re)existir como profissionais." 
			\end{alertblock}
			
			\uncover<2->{
				\begin{alertblock}{Metodologias em diálogo com saberes}
					\justifying
					\textbf{Promover coexistência da pluralidade epistêmica.} "Ainda que um médico indígena seja contemporâneo de um cirurgião de Harvard, ele seria classificado como produtor de um conhecimento 'pré-científico'."
				\end{alertblock}
			}
		\end{frame}
		
		\begin{frame}{Da crítica à insurgência}
			\begin{alertblock}{Matemática como desobediência político-epistêmica}
				\justifying
				\textbf{Subverter o uso da matemática a serviço da colonialidade.} "Colocar a matemática em um plano coletivo e a serviço de lutas sociais, ontológicas e epistêmicas."
			\end{alertblock}
			
			\uncover<2->{
				\begin{alertblock}{Pergunta final que nos convoca.}
					\justifying
					Nós, então, quem somos? Os que reforçam a matemática como metanarrativa central da modernidade? Ou os que subvertem as mesmas peças do projeto colonial?
				\end{alertblock}
			}
		\end{frame}
